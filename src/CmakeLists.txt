cmake_minimum_required (VERSION 3.5)
project (Collision)

set(Boost_USE_STATIC_LIBS OFF)
set(Boost_USE_MULTITHREADED ON)
set(Boost_USE_STATIC_RUNTIME OFF)
find_package(Boost)
include_directories(${Boost_INCLUDE_DIRS})


find_package(Eigen3 REQUIRED)
include_directories(EIGEN3_INCLUDE_DIR)

include_directories(${PYTHON_INCLUDE_PATH})
LINK_DIRECTORIES(${DPYTHON_LIBRARY})
OPTION(BUILD_SHARED_LIBS ON)


ADD_LIBRARY(Collision Collision)
SET_TARGET_PROPERTIES(Collision PROPERTIES  IMPORTED_LINK_INTERFACE_LIBRARIES dbook)
SET_TARGET_PROPERTIES(Collision PROPERTIES LINKER_LANGUAGE C)
